   364→function scrollToId(id: string) {

   366→  const el = document.getElementById(id);
   367→  if (el) el.scrollIntoView({ behavior: "smooth", block: "start" });

   375→  const [shrunk, setShrunk] = useState(false);
   376→  const [open, setOpen] = useState(false);

   381→    <motion.header
   382→      initial={{ y: -40, opacity: 0 }}
   383→      animate={{ y: 0, opacity: 1 }}

   392→      >
   393→        {/* Brand */}
   394→        <button
   395→          onClick={() => scrollToId("top")}

   491→function ScoreRing({ score = 87 }: { score?: number }) {
   492→  const r = 28;
   493→  const c = 2 * Math.PI * r;
   494→  const offset = c * (1 - score / 100);
   495→  return (
   496→    <svg viewBox="0 0 80 80" className="size-20">
   497→      <circle cx="40" cy="40" r={r} fill="none" stroke="currentColor" strokeWidth="6" className="text-border" />
   498→      <motion.circle
   499→        cx="40"
   500→        cy="40"
   501→        r={r}
   502→        fill="none"
   503→        stroke="url(#scoreGrad)"
   504→        strokeWidth="6"
   505→        strokeLinecap="round"
   506→        strokeDasharray={c}
   507→        initial={{ strokeDashoffset: c }}
   508→        animate={{ strokeDashoffset: offset }}
   509→        transition={{ duration: 1.6, delay: 0.6, ease: [0.16, 1, 0.3, 1] }}
   510→        transform="rotate(-90 40 40)"
   511→      />
   512→      <defs>
   513→        <linearGradient id="scoreGrad" x1="0" y1="0" x2="1" y2="1">
   514→          <stop offset="0%" stopColor="oklch(0.62 0.2 258)" />
   515→          <stop offset="100%" stopColor="oklch(0.5 0.18 196)" />
   516→        </linearGradient>
   517→      </defs>
   518→      <text x="40" y="44" textAnchor="middle" className="fill-foreground text-[15px] font-semibold">
   519→        {score}
   520→      </text>
   521→    </svg>
   522→  );
   523→}
